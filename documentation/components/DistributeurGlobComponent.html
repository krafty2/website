  selector: 'app-distributeur-glob',
  standalone: true,
  imports: [
    RouterOutlet,
    MatButtonModule,
    MatIconModule,
    MatSidenavModule,
    SideBarDistribComponent,
  ],
  templateUrl: './distributeur-glob.component.html',
  styleUrl: './distributeur-glob.component.css',
})
export class DistributeurGlobComponent {
  gerant =false;

  mode:any = "side";

  ngOnInit(){
    if(!this.gerant){
      this.mode = "over";
    }
  }
}
